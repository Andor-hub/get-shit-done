Create `project.yml` (xcodegen spec) in the project root with these exact settings:

```yaml
name: HabitX
options:
  bundleIdPrefix: com.habitx
  deploymentTarget:
    iOS: "17.0"
  xcodeVersion: "16.3"
  defaultConfig: Debug
settings:
  base:
    SWIFT_VERSION: "6.0"
    SWIFT_STRICT_CONCURRENCY: complete
targets:
  HabitX:
    type: application
    platform: iOS
    sources:
      - path: HabitX/HabitX
    settings:
      base:
        PRODUCT_BUNDLE_IDENTIFIER: com.habitx.app
        INFOPLIST_KEY_UIApplicationSceneManifest_Generation: true
        INFOPLIST_KEY_UIApplicationSupportsIndirectInputEvents: true
        INFOPLIST_KEY_UILaunchScreen_Generation: true
        INFOPLIST_KEY_UISupportedInterfaceOrientations_iPhone: "UIInterfaceOrientationPortrait"
        CODE_SIGN_ENTITLEMENTS: HabitX/HabitX/HabitX.entitlements
    entitlements:
      path: HabitX/HabitX/HabitX.entitlements
      properties:
        com.apple.security.application-groups:
          - group.com.habitx.shared
  HabitXWidget:
    type: app-extension
    platform: iOS
    product: HabitXWidget
    sources:
      - path: HabitX/HabitXWidget
    settings:
      base:
        PRODUCT_BUNDLE_IDENTIFIER: com.habitx.app.widget
        CODE_SIGN_ENTITLEMENTS: HabitX/HabitXWidget/HabitXWidgetEntitlements.entitlements
        INFOPLIST_KEY_CFBundleDisplayName: HabitXWidget
        INFOPLIST_KEY_NSExtension_NSExtensionPointIdentifier: com.apple.widgetkit-extension
    entitlements:
      path: HabitX/HabitXWidget/HabitXWidgetEntitlements.entitlements
      properties:
        com.apple.security.application-groups:
          - group.com.habitx.shared
    dependencies:
      - target: HabitX
        embed: false
        link: false
```

Create `HabitX/HabitXWidget/HabitXWidget.swift`:
```swift
import WidgetKit
import SwiftUI

struct HabitWidgetEntry: TimelineEntry {
    let date: Date
}

struct HabitTimelineProvider: TimelineProvider {
    func placeholder(in context: Context) -> HabitWidgetEntry {
        HabitWidgetEntry(date: .now)
    }

    func getSnapshot(in context: Context, completion: @escaping (HabitWidgetEntry) -> Void) {
        completion(HabitWidgetEntry(date: .now))
    }

    func getTimeline(in context: Context, completion: @escaping (Timeline<HabitWidgetEntry>) -> Void) {
        let entry = HabitWidgetEntry(date: .now)
        let timeline = Timeline(entries: [entry], policy: .never)
        completion(timeline)
    }
}

struct HabitXWidgetEntryView: View {
    var entry: HabitWidgetEntry

    var body: some View {
        Text("HabitX")
    }
}

struct HabitXWidget: Widget {
    let kind: String = "HabitXWidget"

    var body: some WidgetConfiguration {
        StaticConfiguration(kind: kind, provider: HabitTimelineProvider()) { entry in
            HabitXWidgetEntryView(entry: entry)
        }
        .configurationDisplayName("HabitX")
        .description("Track your daily habits.")
        .supportedFamilies([.systemSmall, .systemMedium])
    }
}
```
